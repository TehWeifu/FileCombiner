class Inhalt {
public:
 Inhalt(const char* s) {
 this->s = new char [strlen(s)+1];
 strcpy(this->s, s);
 }
private:
 char* s;
};

class Container {
public:
 Container(int n) {
 groesse = n;
 tab = new Inhalt[groesse];
 }
 Container(const Container c) {
 groesse = c.groesse;
 tab = new Inhalt[groesse];
 for (int i = 0; i < groesse; i++)
 tab[i] = c.tab[i];
 return *this;
 }
 Inhalt& operator[] (int i) const { return tab[i]; }
private:
 Container& operator=(const Container&);
 Inhalt* tab;
 const int groesse;
};
